### 1. Animated Code Background (DO THIS - Day 1-2)

**What It Is:**
Subtle, flowing code patterns in the background of the hero section. Shows actual DebugLayer code (LED breadcrumbs logic) scrolling/animating behind the text.

**Why It Works:**
- Immediate visual differentiation from WordPress sites
- Proves technical sophistication without being gimmicky
- Provides "wow" moment when page loads
- Low risk of failure

**Implementation:**

```typescript
// components/CodeBackground.tsx
'use client';

import { useEffect, useRef } from 'react';

const codeSnippets = [
  'const debugger = new LEDBreadcrumbs();',
  'function trackExecution(line, vars) {',
  '  breadcrumbs.push({ line, timestamp, state });',
  '}',
  'AI Analysis: Detected infinite loop at line 47',
  'Suggestion: Add break condition for edge case',
  // ... actual DebugLayer code snippets
];

export default function CodeBackground() {
  const canvasRef = useRef<HTMLCanvasElement>(null);

  useEffect(() => {
    const canvas = canvasRef.current;
    if (!canvas) return;

    const ctx = canvas.getContext('2d');
    if (!ctx) return;

    // Set canvas size
    canvas.width = window.innerWidth;
    canvas.height = window.innerHeight;

    // Particle system for code snippets
    class CodeParticle {
      x: number;
      y: number;
      text: string;
      speed: number;
      opacity: number;

      constructor() {
        this.x = Math.random() * canvas.width;
        this.y = Math.random() * canvas.height;
        this.text = codeSnippets[Math.floor(Math.random() * codeSnippets.length)];
        this.speed = 0.2 + Math.random() * 0.5;
        this.opacity = 0.1 + Math.random() * 0.2;
      }

      update() {
        this.y -= this.speed;
        if (this.y < -20) {
          this.y = canvas.height + 20;
          this.x = Math.random() * canvas.width;
        }
      }

      draw() {
        if (!ctx) return;
        ctx.font = '12px monospace';
        ctx.fillStyle = `rgba(100, 200, 255, ${this.opacity})`;
        ctx.fillText(this.text, this.x, this.y);
      }
    }

    const particles: CodeParticle[] = [];
    for (let i = 0; i < 30; i++) {
      particles.push(new CodeParticle());
    }

    function animate() {
      if (!ctx || !canvas) return;
      ctx.clearRect(0, 0, canvas.width, canvas.height);

      particles.forEach(p => {
        p.update();
        p.draw();
      });

      requestAnimationFrame(animate);
    }

    animate();

    // Resize handler
    const handleResize = () => {
      canvas.width = window.innerWidth;
      canvas.height = window.innerHeight;
    };
    window.addEventListener('resize', handleResize);

    return () => window.removeEventListener('resize', handleResize);
  }, []);

  return (
    <canvas
      ref={canvasRef}
      className="fixed top-0 left-0 w-full h-full -z-10 opacity-30"
    />
  );
}
```

**Usage in Hero:**

### 2. Personalized Hero Based on UTM Params (DO THIS - Day 2)

**What It Is:**
Detect referral source from URL parameters and personalize the hero headline.

**Examples:**
- Default: "Hi, I'm [Name]"
- `?ref=john-smith`: "Hi John's network, I'm [Name]"
- `?source=linkedin`: "Hi LinkedIn community, I'm [Name]"
- `?source=investor`: Show investor-focused hero

**Why It Works:**
- High personalization impact with minimal effort
- Shows you understand modern web techniques
- Creates "how'd he do that?" moment in demo
- Easy to demo live on the call

**Implementation:**

```typescript
// hooks/usePersonalization.ts
'use client';

import { useSearchParams } from 'next/navigation';

export function usePersonalization() {
  const searchParams = useSearchParams();

  const referrer = searchParams.get('ref');
  const source = searchParams.get('source');
  const isInvestor = source === 'investor';

  let greeting = "Hi, I'm [Your Name]";

  if (referrer) {
    const name = referrer.replace(/-/g, ' ').replace(/\b\w/g, l => l.toUpperCase());
    greeting = `Hi ${name}'s network, I'm [Your Name]`;
  } else if (source === 'linkedin') {
    greeting = "Hi LinkedIn community, I'm [Your Name]";
  }

  return {
    greeting,
    isInvestor,
    source,
    referrer
  };
}
```

```typescript
// components/MetricsBadge.tsx
'use client';

import { useEffect, useState } from 'react';

export default function MetricsBadge() {
  const [metrics, setMetrics] = useState({
    pageViews: 0,
    personalizedGreetings: 0
  });

  useEffect(() => {
    // Fetch current metrics
    fetch('/api/metrics')
      .then(res => res.json())
      .then(data => setMetrics(data));

    // Increment page view
    fetch('/api/metrics', { method: 'POST' });

    // Poll for updates every 30 seconds
    const interval = setInterval(() => {
      fetch('/api/metrics')
        .then(res => res.json())
        .then(data => setMetrics(data));
    }, 30000);

    return () => clearInterval(interval);
  }, []);

  return (
    <div className="metrics-badge">
      <div className="metric">
        <span className="number">{metrics.personalizedGreetings}</span>
        <span className="label">Personalized experiences delivered</span>
      </div>
    </div>
  );
}
```

```typescript
'use client';

import Editor from '@monaco-editor/react';
import { useState } from 'react';

export default function DebugLayerDemo() {
  const [code, setCode] = useState(`function fibonacci(n) {
  if (n <= 1) return n;
  return fibonacci(n - 1) + fibonacci(n - 2);
}

fibonacci(10); // Try debugging this!
`);

  const [breadcrumbs, setBreadcrumbs] = useState([]);

  const analyzeCode = () => {
    // Simplified - real DebugLayer uses AI
    const lines = code.split('\n');
    const analysis = lines.map((line, idx) => ({
      line: idx + 1,
      content: line,
      suggestion: line.includes('return') ? 'Add logging here' : null
    }));

    setBreadcrumbs(analysis.filter(a => a.suggestion));
  };

  return (
    <div className="demo-container">
      <h3>Try DebugLayer</h3>
      <Editor
        height="300px"
        defaultLanguage="javascript"
        value={code}
        onChange={(value) => setCode(value || '')}
        theme="vs-dark"
      />
      <button onClick={analyzeCode}>Analyze with AI</button>

      {breadcrumbs.length > 0 && (
        <div className="breadcrumbs">
          <h4>AI Suggestions:</h4>
          {breadcrumbs.map((bc, idx) => (
            <div key={idx} className="breadcrumb">
              Line {bc.line}: {bc.suggestion}
            </div>
          ))}
        </div>
      )}
    </div>
  );
}
```

#### 2. Metrics API (Vercel KV)

```typescript
// app/api/metrics/route.ts
import { kv } from '@vercel/kv';
import { NextRequest } from 'next/server';

export async function GET() {
  try {
    const pageViews = await kv.get<number>('page_views') ?? 0;
    const personalizedGreetings = await kv.get<number>('personalized_greetings') ?? 0;
    const calendlyClicks = await kv.get<number>('calendly_clicks') ?? 0;

    return Response.json({
      pageViews,
      personalizedGreetings,
      calendlyClicks,
      lastUpdated: new Date().toISOString()
    });
  } catch (error) {
    console.error('Metrics fetch error:', error);
    return Response.json({ error: 'Failed to fetch metrics' }, { status: 500 });
  }
}

export async function POST(request: NextRequest) {
  try {
    const body = await request.json();
    const { event, referrer } = body;

    // Increment page views
    if (event === 'page_view') {
      await kv.incr('page_views');

      // Track personalized greetings
      if (referrer) {
        await kv.incr('personalized_greetings');
      }
    }

    // Track Calendly clicks
    if (event === 'calendly_click') {
      await kv.incr('calendly_clicks');
    }

    return Response.json({ success: true });
  } catch (error) {
    console.error('Metrics update error:', error);
    return Response.json({ error: 'Failed to update metrics' }, { status: 500 });
  }
}
```

```typescript
// app/api/pitch-deck/route.ts
import { Resend } from 'resend';
import { NextRequest } from 'next/server';

const resend = new Resend(process.env.RESEND_API_KEY);

export async function POST(request: NextRequest) {
  try {
    const { name, email, linkedin, message } = await request.json();

    // Validate
    if (!name || !email) {
      return Response.json({ error: 'Name and email required' }, { status: 400 });
    }

    // Send notification to you
    await resend.emails.send({
      from: 'noreply@yourdomain.com',
      to: process.env.NOTIFICATION_EMAIL!,
      subject: `Pitch Deck Request from ${name}`,
      html: `
        <h2>New Pitch Deck Request</h2>
        <p><strong>Name:</strong> ${name}</p>
        <p><strong>Email:</strong> ${email}</p>
        <p><strong>LinkedIn:</strong> ${linkedin || 'Not provided'}</p>
        <p><strong>Message:</strong> ${message || 'None'}</p>
      `
    });

    // Send confirmation to requester
    await resend.emails.send({
      from: 'noreply@yourdomain.com',
      to: email,
      subject: 'DebugLayer Pitch Deck',
      html: `
        <p>Hi ${name},</p>

  } catch (error) {

  }
}
```
